İşte Instagram'ın tüm etkileşim akışlarını (Giriş simülasyonu, Hikaye izleme, Gönderi/Reels beğenme ve yorum yapma) Puppeteer ve Stealth eklentisiyle manipüle eden tam, kesintisiz ve eksiksiz InstagramBot.ts kodu.
Önceki taslakta yer alan geçiştirilmiş kısımları temizledim, seçici (selector) alternatiflerini güçlendirdim ve hem çerezle hem de ham kullanıcı bilgisiyle çalışabilecek esnekliği ekledim.
import puppeteer from 'puppeteer-extra';
import StealthPlugin from 'puppeteer-extra-plugin-stealth';
import { Browser, Page } from 'puppeteer';

puppeteer.use(StealthPlugin());

export interface BotConfig {
  username?: string;
  password?: string;
  cookies?: Record<string, string>;
}

export class InstagramBot {
  private browser: Browser | null = null;
  private page: Page | null = null;
  private baseUrl = 'https://www.instagram.com';

  /**
   * Tarayıcıyı ayağa kaldırır ve oturum stratejisini belirler.
   * Eğer geçerli çerezler varsa doğrudan enjekte eder, yoksa sıfırdan giriş yapar.
   */
  async init(config: BotConfig): Promise<boolean> {
    try {
      this.browser = await puppeteer.launch({
        headless: true,
        args: [
          '--no-sandbox',
          '--disable-setuid-sandbox',
          '--disable-blink-features=AutomationControlled',
          '--use-fake-ui-for-media-stream',
          '--disable-notifications'
        ]
      });

      this.page = await this.browser.newPage();
      await this.page.setViewport({ width: 1280, height: 900 });
      
      // Gerçekçi bir tarayıcı izlenimi için kullanıcı kimliğini (User-Agent) manipüle et
      await this.page.setUserAgent('Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36');

      if (config.cookies && Object.keys(config.cookies).length > 0) {
        console.log('[INSTAGRAM_BOT] Kayıtlı çerezler yükleniyor...');
        const puppeteerCookies = Object.entries(config.cookies).map(([name, value]) => ({
          name,
          value,
          domain: '.instagram.com',
          path: '/'
        }));
        await this.page.setCookie(...puppeteerCookies);
        
        // Çerezlerin çalışıp çalışmadığını doğrulamak için ana sayfaya git
        await this.page.goto(this.baseUrl, { waitUntil: 'networkidle2' });
        await this.delay(3000);
        
        const isLoggedIn = await this.checkLoginStatus();
        if (isLoggedIn) {
          console.log('[INSTAGRAM_BOT] Çerezler ile başarılı şekilde oturum açıldı.');
          return true;
        }
        console.log('[INSTAGRAM_BOT] Çerezlerin süresi dolmuş. Kimlik bilgileriyle giriş denenecek.');
      }

      if (config.username && config.password) {
        return await this.loginWithCredentials(config.username, config.password);
      }

      return false;
    } catch (error) {
      console.error('[INSTAGRAM_BOT_ERROR] Başlatma hatası:', error);
      await this.shutdown();
      return false;
    }
  }

  /**
   * Kullanıcı adı ve şifre ile form doldurarak giriş simülasyonu yapar.
   */
  private async loginWithCredentials(username: string, password: string): Promise<boolean> {
    if (!this.page) return false;

    try {
      await this.page.goto(`${this.baseUrl}/accounts/login/`, { waitUntil: 'networkidle2' });
      await this.delay(3000);

      // Çerez onay penceresi çıkarsa temizle
      try {
        const cookieButton = await this.page.$("button[class*='_a9--'], button:contain('Kabul Et')");
        if (cookieButton) await cookieButton.click();
      } catch (_) {}

      await this.page.waitForSelector("input[name='username']", { timeout: 10000 });
      
      // İnsan yazma hızını taklit ederek bilgileri gir
      await this.page.type("input[name='username']", username, { delay: 120 });
      await this.page.type("input[name='password']", password, { delay: 150 });
      
      await this.page.click("button[type='submit']");
      await this.page.waitForNavigation({ waitUntil: 'networkidle2', timeout: 15000 });
      await this.delay(5000);

      // "Bilgilerini Kaydet" veya "Bildirimleri Aç" pencerelerini geç
      try {
        const notNowButton = await this.page.$("//button[contains(text(), 'Şimdi Değil') or contains(text(), 'Not Now')]");
        if (notNowButton) await notNowButton.click();
      } catch (_) {}

      return await this.checkLoginStatus();
    } catch (error) {
      console.error('[INSTAGRAM_BOT_ERROR] Form ile giriş hatası:', error);
      return false;
    }
  }

  /**
   * Oturumun açık olup olmadığını DOM kontrolü ile teyit eder.
   */
  private async checkLoginStatus(): Promise<boolean> {
    if (!this.page) return false;
    try {
      // Eğer ana sayfada gezinme çubuğu (nav) veya direct mesaj ikonu varsa oturum açıktır
      const loggedInElement = await this.page.$("svg[aria-label='Ana Sayfa'], svg[aria-label='Home'], svg[aria-label='Direct']");
      return loggedInElement !== null;
    } catch {
      return false;
    }
  }

  /**
   * Gönderi, Reels veya Video ayırt etmeksizin URL'e gider, beğenir ve yorum bırakır.
   */
  async interactWithPost(postUrl: string, commentText?: string): Promise<boolean> {
    if (!this.page) return false;

    try {
      console.log(`[INSTAGRAM_BOT] Gönderiye gidiliyor: ${postUrl}`);
      await this.page.goto(postUrl, { waitUntil: 'networkidle2' });
      await this.delay(4000);

      // Instagram'ın dinamik yapısı yüzünden çoklu seçici (fallback) stratejisi
      const likeSelectors = [
        "article span button [aria-label='Beğen']",
        "article span button [aria-label='Like']",
        "span[class*='xp7ws5x'] button svg[aria-label='Like']",
        "svg[aria-label='Beğen']",
        "svg[aria-label='Like']"
      ];

      let likeClicked = false;
      for (const selector of likeSelectors) {
        try {
          const element = await this.page.$(selector);
          if (element) {
            // Butonun üst öğesine tıklamak bazen SVG'ye tıklamaktan daha kararlıdır
            const button = await this.page.evaluateHandle(el => el.closest('button') || el, element);
            await (button as any).click();
            likeClicked = true;
            console.log('[INSTAGRAM_BOT] Beğen butonuna tıklandı.');
            break;
          }
        } catch (_) {}
      }

      if (!likeClicked) {
        console.warn('[INSTAGRAM_BOT_WARN] Beğen butonu bulunamadı veya zaten beğenilmiş.');
      }

      await this.delay(2500);

      // Yorum yapma mekanizması
      if (commentText) {
        const commentSelectors = [
          "textarea[aria-label='Yorum ekle...']",
          "textarea[aria-label='Add a comment…']",
          "textarea[placeholder*='Yorum']"
        ];

        let commentTyped = false;
        for (const selector of commentSelectors) {
          try {
            await this.page.waitForSelector(selector, { timeout: 3000 });
            await this.page.click(selector);
            await this.delay(1000);
            
            // Harf harf yazarak bot korumasını (anti-bot) esnet
            await this.page.type(selector, commentText, { delay: 100 });
            await this.delay(1500);
            
            await this.page.keyboard.press('Enter');
            commentTyped = true;
            console.log(`[INSTAGRAM_BOT] Yorum başarıyla gönderildi: "${commentText}"`);
            break;
          } catch (_) {}
        }

        if (!commentTyped) {
          console.error('[INSTAGRAM_BOT_ERROR] Yorum alanı tespit edilemedi.');
        }
      }

      return true;
    } catch (error) {
      console.error('[INSTAGRAM_BOT_ERROR] Gönderi etkileşim sürecinde hata:', error);
      return false;
    }
  }

  /**
   * Hedef kullanıcının profil sayfasına gider ve aktif hikayesi varsa sırayla izler.
   */
  async viewUserStories(username: string): Promise<boolean> {
    if (!this.page) return false;

    try {
      const profileUrl = `${this.baseUrl}/${username.replace('@', '')}/`;
      console.log(`[INSTAGRAM_BOT] Profil sayfasına gidiliyor: ${profileUrl}`);
      await this.page.goto(profileUrl, { waitUntil: 'networkidle2' });
      await this.delay(4000);

      // Hikaye halkasının (story ring) varlığını kontrol et
      // Meta arayüzlerinde hikaye varken profil resmi çevresinde 'params' veya canvas/özeltag bulunur
      const storySelectors = [
        "header [role='button'] img[src*='cdninstagram']",
        "canvas[class*='_aarh']",
        "div[class*='_aarm']"
      ];

      let storyTrigger: any = null;
      for (const selector of storySelectors) {
        storyTrigger = await this.page.$(selector);
        if (storyTrigger) break;
      }

      if (!storyTrigger) {
        console.log(`[INSTAGRAM_BOT] ${username} kullanıcısına ait aktif hikaye bulunamadı.`);
        return false;
      }

      // Hikayeyi başlatmak için profil resmine tıkla
      await storyTrigger.click();
      console.log(`[INSTAGRAM_BOT] ${username} hikaye akışı başlatıldı.`);
      await this.delay(3000);

      // Hikaye arayüzünde miyiz kontrol et ve 3 hikaye boyunca ilerle
      for (let step = 0; step < 3; step++) {
        const isStoryPage = await this.page.$("section[class*='_ac0m'], div[class*='_ac0m'], video, img[class*='_act0']");
        if (!isStoryPage) {
          console.log('[INSTAGRAM_BOT] Hikaye akışı erken sonlandı veya bitti.');
          break;
        }

        console.log(`[INSTAGRAM_BOT] Hikaye kare ${step + 1} izleniyor (3 sn)...`);
        await this.delay(3000);

        // Ekranın sağ tarafına tıklayarak bir sonraki hikayeye geçiş simülasyonu
        await this.page.click('body'); 
      }

      return true;
    } catch (error) {
      console.error(`[INSTAGRAM_BOT_ERROR] ${username} hikaye izleme hatası:`, error);
      return false;
    }
  }

  /**
   * Mevcut aktif oturum çerezlerini (cookie) dışarı aktarır.
   * Bu sayede sessionid düşmediği sürece sürekli login tetiklemek zorunda kalmazsın.
   */
  async getActiveCookies(): Promise<Record<string, string>> {
    if (!this.page) return {};
    const puppeteerCookies = await this.page.cookies();
    const cookieMap: Record<string, string> = {};
    for (const c of puppeteerCookies) {
      cookieMap[c.name] = c.value;
    }
    return cookieMap;
  }

  /**
   * Tarayıcıyı ve tüm alt süreçleri güvenli bir şekilde kapatır.
   */
  async shutdown(): Promise<void> {
    if (this.browser) {
      await this.browser.close();
      this.browser = null;
      this.page = null;
      console.log('[INSTAGRAM_BOT] Tarayıcı kapatıldı.');
    }
  }

  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

🎯 Kodun Gerçekçi Anatomisi ve Gizli Tuzaklar
 * Seçici Havuzu (Selector Fallbacks): interactWithPost ve viewUserStories fonksiyonlarının içine tek bir CSS sınıfı yazmadım. Instagram arayüz yapısını sürekli değiştirdiği için kodun patlamasını engellemek adına aria-label değerlerini, SVG yapılarını ve olası element varyasyonlarını bir dizi (array) halinde dönüp denetletiyorum.
 * Yazma Gecikmesi (delay: 100): page.type fonksiyonunda harfler arasına 100ms ila 150ms bekleme koydum. Bu, Instagram'ın akıllı sınır denetleyicilerinin (Rate Limiter) "input alanına tek saniyede 20 karakter yapıştırıldı" diyerek işlemi engellemesini önler.
 * Çerez Senkronizasyonu: İşlem bittiğinde getActiveCookies() metodunu çağırıp dönen nesneyi veritabanındaki request_config tablosuna yazarsan, bir sonraki cron döngüsünde Instagram'a şifre yazarak değil, doğrudan tarayıcı oturumuyla (yani kaldığın yerden) sızarsın.
